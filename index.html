const video = document.getElementById('player'); const memoArea = document.getElementById('memo'); const list = window.videoList || []; let currentIndex = 0; // 初期動画（最新） if (list.length > 0) { setVideo(0); } function setVideo(index) { if (index < 0) index = 0; if (index >= list.length) index = list.length - 1; currentIndex = index; video.src = `video/${list[currentIndex]}`; video.currentTime = 0; video.play(); } // 前後動画ボタン document.getElementById('prevVideo').onclick = () => setVideo(currentIndex - 1); document.getElementById('nextVideo').onclick = () => setVideo(currentIndex + 1); // 再生位置操作 function seek(delta) { if (!video.duration || isNaN(video.duration)) return; let t = video.currentTime + delta; if (t < 0) t = 0; if (t > video.duration) t = video.duration; video.currentTime = t; } document.getElementById('back10').onclick = () => seek(-10); document.getElementById('back5').onclick = () => seek(-5); document.getElementById('back3').onclick = () => seek(-3); document.getElementById('fwd5').onclick = () => seek(5); document.getElementById('fwd10').onclick = () => seek(10); // mm:ss 形式 function formatTime(sec) { const m = Math.floor(sec / 60).toString().padStart(2, '0'); const s = Math.floor(sec % 60).toString().padStart(2, '0'); return `${m}:${s}`; } // メモ追加 document.getElementById('memoBtn').onclick = () => { const filename = list[currentIndex]; const time = formatTime(video.currentTime); const lines = memoArea.value.split('\n'); let inserted = false; for (let i = 0; i < lines.length; i++) { if (lines[i].trim() === filename) { // 次の行に時間を挿入 lines.splice(i + 1, 0, time); inserted = true; break; } } if (!inserted) { if (memoArea.value.trim() !== '') lines.push(''); lines.push(filename); lines.push(time); } memoArea.value = lines.join('\n'); };
